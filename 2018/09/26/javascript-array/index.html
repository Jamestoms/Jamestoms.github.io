对象数组去重
jquery方法

function uniqueArr1(arr) {
  const newArr = [];
  $.each(arr, function(i, v) {
    var flag = true;
    if (newArr.length > 0) {
      $.each(newArr, function(n, m) {
        if (newArr[n].contact_id == arr[i].contact_id) {
          flag = false;
        }
      });
    }
    if (flag) {
      newArr.push(arr[i]);
    }
  });
  return newArr;
}
